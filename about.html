Servidor

server <- function(input, output, session) { # Datos reactivos para almacenar estudiantes datos <- reactiveValues(registro = data.frame(Nombre = character(), Cedula = character(), Email = character(), Carrera = character(), stringsAsFactors = FALSE))

# Función para registrar estudiantes observeEvent(input\(registrar, { if (input\)nombre != “” && input\(cedula != "" && input\)email != ““) { datos\(registro <- rbind(datos\)registro, data.frame( Nombre = input\(nombre, Cedula = input\)cedula, Email = input\(email, Carrera = input\)carrera, stringsAsFactors = FALSE )) showNotification(”Registro completado exitosamente.”, type = “message”) } else { showNotification(“Por favor, complete todos los campos obligatorios.”, type = “error”) } })

# Mostrar tabla de estudiantes registrados output\(tabla_registro <- renderDT({ datatable(datos\)registro, options = list(pageLength = 5)) })

# Guardar progreso del proceso de graduación progreso <- reactiveValues(etapas = NULL)

observeEvent(input\(guardar_progreso, { progreso\)etapas <- input$etapas showNotification(“Progreso guardado.”, type = “message”) })

# Mostrar progreso del estudiante output\(progreso <- renderPrint({ if (is.null(progreso\)etapas)) { “No se ha completado ninguna etapa.” } else { paste(“Etapas completadas:”, paste(progreso$etapas, collapse = “,”)) } }) }

Correr la aplicación

shinyApp(ui, server)
